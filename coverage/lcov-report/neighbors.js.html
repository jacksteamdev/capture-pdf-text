/**
 * Get object with padded boundary properties
 *
 * padItemBy :: (a -> number) -> a -> {left, right, bottom, top}
 */
export const padItem = curry((padFn, item) => {
  const pad = padFn(item)
  return {
    left: item.left - pad,
    right: item.right + pad,
    bottom: item.bottom - pad,
